    <script>
        // Add smooth interactions and animations
        document.addEventListener('DOMContentLoaded', function() {
            // Album card hover effects
            const albumCards = document.querySelectorAll('.album-card');
            albumCards.forEach(card => {
                card.addEventListener('mouseenter', function() {
                    this.style.transform = 'translateY(-10px) scale(1.02)';
                });
                
                card.addEventListener('mouseleave', function() {
                    this.style.transform = 'translateY(0) scale(1)';
                });
            });

            // Search focus effects
            const searchInput = document.querySelector('.search-input');
            searchInput.addEventListener('focus', function() {
                this.parentElement.style.transform = 'scale(1.02)';
            });
            
            searchInput.addEventListener('blur', function() {
                this.parentElement.style.transform = 'scale(1)';
            });

            // Floating button animation
            const floatingBtn = document.querySelector('.floating-add');
            floatingBtn.addEventListener('click', function() {
                this.style.transform = 'scale(0.9) rotate(90deg)';
                setTimeout(() => {
                    this.style.transform = 'scale(1.1) rotate(90deg)';
                }, 100);
                setTimeout(() => {
                    this.style.transform = 'scale(1) rotate(0deg)';
                }, 300);
            });

            // Stagger animation for album cards
            albumCards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(30px)';
                
                setTimeout(() => {
                    card.style.transition = 'all 0.6s ease';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 100);
            });
        });
    </script>
